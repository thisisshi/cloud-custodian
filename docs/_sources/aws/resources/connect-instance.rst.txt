.. _aws.connect-instance.actions.set-attributes:

set-attributes
++++++++++++++
Set the attributes for the connect resources

:example:

.. code-block:: yaml

    policies:
      - name: connect-set-contact-lens
        resource: connect-instance
        filters:
          - type: instance-attribute
            key: Attribute.Value
            value: false
            attribute_type: CONTACT_LENS
        actions:
          - type: set-attributes
            attribute_type: CONTACT_LENS
            value: true
      - name: connect-disable-contact-lens
        resource: connect-instance
        filters:
          - type: instance-attribute
            key: Attribute.Value
            value: true
            attribute_type: CONTACT_LENS
        actions:
          - type: set-attributes
            attribute_type: CONTACT_LENS
            value: false
