credential
++++++++++

    - name: cam-user-access-key-rotation
      resource: tencentcloud.cam-user
      filters:
      - type: credential
        key: access_keys.CreateTime
        value_type: age
        value: 3
        op: greater-than
      - type: credential
        key: access_keys.Status
        value: Active
    - name: cam-user-mfa-missing
      resource: tencentcloud.cam-user
      filters:
      - type: credential
        key: ConsoleLogin
        value: 1
